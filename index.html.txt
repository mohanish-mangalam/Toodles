<script>
let gallery=document.getElementById("gallery");
for(let i=1;i<=10;i++){
gallery.innerHTML+=`
<div class="photo-card">
<img src="photo${i}.jpg">
<div class="caption">Edit this caption ${i} 😌</div>

document.getElementById("startBtn").onclick=function(){
document.getElementById("music").play();
document.getElementById("startScreen").style.display="none";
gallery.style.display="block";
document.getElementById("proposal").style.display="block";
showOwl("Hoot hoot Toodles… scroll slowly ❤️");
};

function showOwl(text){
let bubble=document.getElementById("owlBubble");
bubble.innerText=text;
bubble.style.display="block";
setTimeout(()=>bubble.style.display="none",4000);
}

let noBtn=document.getElementById("noBtn");
let attempts=0;
noBtn.style.top="70%";
noBtn.style.left="50%";

noBtn.addEventListener("touchstart",escape);
noBtn.addEventListener("mouseover",escape);

function escape(){
attempts++;
let x=Math.random()*(window.innerWidth-100);
let y=Math.random()*(window.innerHeight-100);
noBtn.style.left=x+"px";
noBtn.style.top=y+"px";

if(attempts==3) showOwl("Ma’am… destiny is buffering.");
if(attempts==4){
document.getElementById("shivaContainer").style.display="block";
}
if(attempts==5){
noBtn.innerText="Accha baba thoda sochti hu";
}
}

document.getElementById("yesBtn").onclick=function(){
document.getElementById("wedding").style.display="block";
confetti();
};

function confetti(){
for(let i=0;i<150;i++){
let div=document.createElement("div");
div.className="heart";
div.innerHTML="❤️";
div.style.left=Math.random()*window.innerWidth+"px";
div.style.top=Math.random()*window.innerHeight+"px";
document.body.appendChild(div);
setTimeout(()=>div.remove(),1000);
}
}

/* Heart trail on touch */
document.addEventListener("touchmove",function(e){
let heart=document.createElement("div");
heart.className="heart";
heart.innerHTML="💖";
heart.style.left=e.touches[0].clientX+"px";
heart.style.top=e.touches[0].clientY+"px";
document.body.appendChild(heart);
setTimeout(()=>heart.remove(),1000);
});
